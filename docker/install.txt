atualizar o sistema
    sudo apt update


pacotes de pré-requisitos para permitir ao APT o uso de pacotes seguros HTTPS
    sudo apt install apt-transport-https ca-certificates curl software-properties-common


chave GPG para garantir a validade dos pacotes vindos do repositório do Docker
    curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg


adicionar o repositório Docker as fontes de pacotes do Ubuntu
    echo "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null


atualizar o sistema
    sudo apt update


instalar o docker
    sudo apt install docker-ce


verificar instalação
    sudo systemctl status docker
    docker --version


adicionar usuário ao grupo 'docker'
    sudo usermod -aG docker ${USER}
    su - ${USER}


listar conteiners em execução
    docker ps -a
    (-a lista os inativos)


remover/inserir configuração de inicio automático de um conteiner
    docker update --restart=no nome_do_conteiner
    docker update --restart=always nome_do_conteiner


iniciar/parar um conteiner
    docker start nome_do_container
    docker stop nome_do_container


remover um conteiner
    docker stop nome_do_conteiner
    docker rm nome_do_conteiner
